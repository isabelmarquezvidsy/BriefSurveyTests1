                    console.warn("Custom token sign-in failed, attempting anonymous sign-in:", customTokenError);

                    // If custom token fails, try anonymous sign-in as a fallback

                    await signInAnonymously(auth);

                  }

                } else {

                  // If no custom token is provided or it's empty, sign in anonymously

                  await signInAnonymously(auth);

                }
